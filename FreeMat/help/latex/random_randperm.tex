\section{RANDPERM Random permutation}

\subsection{USAGE}

\begin{verbatim}
   y = randperm(n)
\end{verbatim}
\verb|y| is a random permutation of integers from 1 to \verb|n|.
\verb|randperm| calls \verb|rand| and changes its state.
\subsection{Example}

\begin{verbatim}
--> y = randperm(10)

y = 
  2  5 10  8  1  6  3  7  9  4 
\end{verbatim}
